class Person:
def __init__(self,name):
    self.name=name
        
class LikesCricket:
    def __init__(self,person):
        self.person=person
        
def does_sachin_like_cricket(people):
    for person in people:
        if isinstance(person,LikesCricket):
            return True
    return False

sachin = Person("Sachin")
virat = Person("Virat")
dhoni = LikesCricket(Person("Dhoni"))

people = [sachin,virat,dhoni]

if does_sachin_like_cricket(people):
    print("Sachin likes Cricket")
else:
    print("Sachin doesn't like Cricket")
